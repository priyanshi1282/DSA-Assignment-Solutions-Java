solution :

class Solution {
    public boolean checkIfExist(int[] arr) {
        int n=arr.length;
        for(int i=0;i<n-1;i++){
                for(int j=i+1;j<n;j++)
                     if(arr[j]== 2*arr[i] || 2*arr[j]==arr[i])
                    return true;
                
        }
        
        return false;
        
    }
}
